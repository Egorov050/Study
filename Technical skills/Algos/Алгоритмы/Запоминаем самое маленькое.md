class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        mb = 0
        bprofit = 0
        for i in range(1, len(prices)) : 
            pr = prices[i] - prices[mb]
            bprofit = max(bprofit, pr)
            if prices[i] < prices[mb] : 
                mb = i
        return bprofit
